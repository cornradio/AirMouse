    <script src="https://cdn.socket.io/4.5.4/socket.io.min.js"></script>
    <script>
        const socket = io();
        const slider = document.getElementById('sens-slider');

        // 1. 从 LocalStorage 读取 Sense，如果没有则默认 4.0 
        let sensitivity = parseFloat(localStorage.getItem('mouse_sense')) || 4.0;
        slider.value = sensitivity;

        // 2. 监听滑动并实时保存 
        slider.oninput = (e) => {
            sensitivity = parseFloat(e.target.value);
            localStorage.setItem('mouse_sense', sensitivity);
        };

        // 鼠标逻辑
        const pad = document.getElementById('pad');
        let lastX, lastY, startX, startY, isDragging = false, hasMoved = false, touchCount = 0;

        pad.addEventListener('touchstart', e => {
            hasMoved = false; touchCount = e.touches.length;
            startX = e.touches[0].clientX; startY = e.touches[0].clientY;
            lastX = startX; lastY = startY;
            if (touchCount === 3) { isDragging = true; socket.emit('drag_start'); pad.style.background = "#000810"; }
        });

        pad.addEventListener('touchmove', e => {
            e.preventDefault();
            const curX = e.touches[0].clientX; const curY = e.touches[0].clientY;
            if (Math.hypot(curX - startX, curY - startY) > 5) hasMoved = true;
            if (e.touches.length === 1 || e.touches.length === 3) {
                socket.emit('move', { dx: (curX - lastX) * sensitivity, dy: (curY - lastY) * sensitivity });
                lastX = curX; lastY = curY;
            } else if (e.touches.length === 2) {
                const dy = curY - lastY;
                if (Math.abs(dy) > 5) { socket.emit('scroll', { dy: dy > 0 ? 1 : -1 }); lastY = curY; }
            }
        }, { passive: false });

        pad.addEventListener('touchend', e => {
            if (e.touches.length === 0) {
                if (isDragging) { socket.emit('drag_end'); isDragging = false; pad.style.background = "#0a0a0a"; }
                else if (!hasMoved) {
                    if (touchCount === 2) socket.emit('click', { button: 'right' });
                    else if (touchCount === 1) socket.emit('click', { button: 'left' });
                }
            }
        });
    </script>
